The data is formatted as clean CSV files, ready for any data science stack.

analysis.py 
import pandas as pd

# Load US state-level data directly from raw source
url = "https://raw.githubusercontent.com/jjoycehe/covid-19-data/master/us-states.csv"
df = pd.read_csv(url)

# Filter for latest date
latest = df.sort_values('date').groupby('state').last().reset_index()

print(latest.head())
