Core Routing Logic

The application uses React Router v6 to handle protected routes and nested views for notes and collections.

import { Routes, Route } from "react-router-dom";
import { Toaster } from "react-hot-toast";

function App() {
  return (
    <ErrorBoundary>
      <AuthProvider>
        <Routes>
          <Route element={<PrivateRoute />}>
            <Route index element={<Notes />} />
            <Route path="collections/:id" element={<Collection />} />
            <Route path="notes/new" element={<NewNote />} />
          </Route>
        </Routes>
      </AuthProvider>
    </ErrorBoundary>
  );
}
