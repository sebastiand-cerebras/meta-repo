Quick Start

terminal — bash — 80x24
$ pip install nql-to-sql
from nql_to_sql import Translator
from nql_to_sql.schema import Table
# Define your schema
users_schema = Table("users", ["id", "name", "email", "created_at"])
translator = Translator([users_schema])
query = translator.translate("Find users named John")
print(query)
# Output: SELECT * FROM users WHERE name = 'John'
